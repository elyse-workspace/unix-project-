RESTORE SCRIPT#!/bin/bash#Test if no name provided or File does not existif [ $# -eq 0 ] ; then     echo "Error: No filename provided!"     exit 1elif [ ! -e $HOME/recyclebin/$1 ] ; then     echo "Error: File does not exist!"     exit 1fi#Restore file to original location from .restore.info #If already exist in target directory, then prompt overwrite.pathname_dir=$(grep $1 $HOME/.restore.info | cut -d ":" -f2)pathname=$(readlink -f $pathname_dir)if [ -e $pathname_dir ] ; then     read -p "Do you want to overwrite? y/n " reply     case $reply in          y|Y|^y*|^Y*) mv -f  $HOME/recyclebin/$1 $pathname                       grep -v $1 $HOME/.restore.info > $HOME/.restemp.info                       mv $HOME/.restemp.info $HOME/.restore.info                       echo "Finished!"                       exit 0;;          *) echo "Finished!"             exit 0;;          esacelse     mv $HOME/recyclebin/$1 $pathname     grep -v $1 $HOME/.restore.info > $HOME/.restemp.info     mv $HOME/.restemp.info $HOME/.restore.info     echo "Finished!"fi
